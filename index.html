  created_at timestamptz DEFAULT now()

  id integer PRIMARY KEY, name text, phone text,
  email text, company text, reward text, game text,
  voucher text, tags text[], svcs text[], note text,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE counters (
  name text PRIMARY KEY, value integer DEFAULT 100
);

CREATE OR REPLACE FUNCTION next_ticket()
RETURNS integer LANGUAGE sql AS $$
  UPDATE counters SET value=value+1
  WHERE name='ticket' RETURNING value;
$$;
